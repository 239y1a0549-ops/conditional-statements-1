6.second largest
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int T = sc.nextInt();
        
        for (int i = 0; i < T; i++) {
            int A = sc.nextInt();
            int B = sc.nextInt();
            int C = sc.nextInt();
            
            if ((A > B && A < C) || (A < B && A > C)) {
                System.out.println(A);
            } else if ((B > A && B < C) || (B < A && B > C)) {
                System.out.println(B);
            } else {
                System.out.println(C);
            }
        }
    }
}

8. fibbonacci number
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int N = sc.nextInt();
        
        int first = 1;
        int second = 1;
        int next = 1;
        
        for (int i = 3; i <= N; i++) {
            next = first + second;
            first = second;
            second = next;
        }
        
        System.out.println(next);
    }
}
